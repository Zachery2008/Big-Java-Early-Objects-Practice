01: /**
02:    A cash register totals up sales and computes change due.
03: */
04: public class CashRegister
05: {
06:    /**
07:       Constructs a cash register with no money in it.
08:    */
09:    public CashRegister()
10:    {
11:       purchase = 0;
12:       payment = 0;
13:    }
14: 
15:    /**
16:       Records the sale of an item.
17:       @param amount the price of the item
18:    */
19:    public void recordPurchase(double amount)
20:    {
21:       double total = purchase + amount;
22:       purchase = total;
23:    }
24: 
25:    /**
26:       Enters the payment received from the customer.
27:       @param amount the amount of the payment
28:    */
29:    public void enterPayment(double amount)
30:    {
31:       payment = amount;
32:    }
33: 
34:    /**
35:       Computes the change due and resets the machine for the next customer.
36:       @return the change due to the customer
37:    */
38:    public double giveChange()
39:    {   
40:       double change = payment - purchase;
41:       purchase = 0;
42:       payment = 0;
43:       return change;
44:    }
45:    private double purchase;
46:    private double payment;
47: }
