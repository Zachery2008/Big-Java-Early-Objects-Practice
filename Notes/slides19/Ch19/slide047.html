The Quicksort Algorithm

private int partition(int from, int to)
{
   int pivot = a[from];
   int i = from - 1;
   int j = to + 1;
   while (i < j)
   {
      i++; while (a[i] < pivot) i++;
      j--; while (a[j] > pivot) j--;
      if (i < j) swap(i, j);
   }
   return j;
}
