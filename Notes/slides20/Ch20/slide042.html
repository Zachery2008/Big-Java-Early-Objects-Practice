The Linked List Iterator's add Method

public void add(Object obj)
{
   if (position == null)
   {
      addFirst(obj);
      position = first;
   }
   else
   {
      Node newNode = new Node();
      newNode.data = obj;
      newNode.next = position.next;  
      position.next = newNode;  
      position = newNode;  
   }
   previous = position; 
}
