Class Pair

public class Pair<T, S>
{
   public Pair(T firstElement, S secondElement)
   {
      first = firstElement;
      second = secondElement;
   }

   public T getFirst() { return first; }
   public S getSecond() { return second; }

   private T first;
   private S second;
}
